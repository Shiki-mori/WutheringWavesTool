## 抽离工具函数-parseGachaUrl—URL参数获取
新建文件`backend/utils/parseUrl.js`：（放置在后端避免用户恶意输入）  
```js
function parseGachaUrl(url){
    try{
        const parsed=new URL(url);//URL是JavaScript内置的类，可以解析URL字符串。new URL(url)会返回一个URL对象，包含了URL的各个部分。

        const params=parsed.searchParams;//searchParams是URL对象的一个属性，是一个URLSearchParams对象，可以用来获取URL中的查询参数。

        return{
            origin:parsed.origin,//origin是URL对象的一个属性，表示URL的协议和主机部分，例如https://example.com
            path:parsed.pathname,//pathname是URL对象的一个属性，表示URL的路径部分，例如/analyze
            query:Object.fromEntries(params.entries())//entries()是URLSearchParams对象的方法，返回一个迭代器，包含了URL中的所有查询参数的键值对。Object.fromEntries()是JavaScript内置的方法，可以把一个键值对数组转换成一个对象。

            //常用字段（可能存在，将在后续测试过程中完善）
            uid:params.get("uid"),//get()是URLSearchParams对象的方法，可以获取指定查询参数的值，例如uid=12345678
            authkey:params.get("authkey")
        };
    }
    catch(err){
        return null;//如果URL无效，new URL(url)会抛出一个错误，我们捕获这个错误并返回null，表示解析失败。
    }
}

module.exports={
    parseGachaUrl
};//导出parseGachaUrl函数，使其可以在其他文件中被引入和使用。
```
在接口中使用分析函数。编辑`index.js`文件： 
```js
const {parseGachaUrl}=require("./utils/parseUrl")//引入解析URL的工具函数

app.get("/api/gacha",(req,res)=>{

  if(!parsed){

```

## 获取URL中的具体字段名
使用浏览器控制台，切换到Network标签->Fetch/XHR->(刷新获取最新query)Name\[query\]->Payload&Preview。着重查看payload&response中的数据。  
payload是请求参数，依此构造请求结构；  
response是返回数据，依此构造返回结构。  
>Headers请求头 作用：浏览器身份信息，鉴权，防爬   
>Payload 请求发给服务器的参数，决定向服务器传什么参数    
>Response 服务器真实返回的数据，是需要被分析的  
>Preview Response的可视化版本，格式化、自动展开JSON

经过测试，以上字段中不存在功能类似于`authkey`的认证参数。`authkey`表示登录态。推测抽卡接口依赖浏览器隐式登录态（cookie）。  
在前端调用该API会被浏览器拦截，因为`aki-game2.com`不允许本地域名跨域访问。必须通过后端进行请求转发（代理）。    
需要改变开发方向：前端直连接口，模拟浏览器。  

## 原理
游戏的抽卡链接是一个静态页面地址。真正的API终点是Header中的`:authority`组合出的地址：`https://gmserver-api.aki-game2.com/gacha/record/query`。  
请求方式：`POST`    

## 后端实现
在后端使用`axios`发起请求，绕过浏览器的同源策略。  
```js
const axios = require('axios')  

app.post('/api/gacha/proxy', async (req, res) => {
    try {
        // 1. 获取前端传来的 Payload
        // 这里的 req.body 应该包含：serverId, playerId, cardPoolId, cardPoolType, languageCode, recordId 等
        const payload = req.body;

        console.log('正在转发请求，Payload:', payload);

        // 2. 发起请求到鸣潮官方接口
        const response = await axios({
            url: 'https://gmserver-api.aki-game2.com/gacha/record/query',
            method: 'POST',
            data: payload,
            headers: {
                // 必须伪造这些头部，模仿游戏内嵌浏览器的行为
                'Content-Type': 'application/json',
                'Accept': 'application/json, text/plain, */*',
                'Origin': 'https://aki-gm-resources.aki-game.com',
                'Referer': 'https://aki-gm-resources.aki-game.com/',
                'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36'
            }
        });

        // 3. 将官方返回的原样传回给前端
        console.log('官方接口响应成功:', response.data.message);
        res.json(response.data);

    } catch (error) {
        console.error('转发失败:', error.response?.data || error.message);
        res.status(500).json({
            error: 'Backend Proxy Error',
            details: error.response?.data || error.message
        });
    }
});
```
api下新建一个parseUrl.ts(目前的项目结构存在一些问题，需要后续整合):
```ts
//从链接中准确提取出官方API认识的字段名
export const parseGachaUrl = (fullUrl: string): {
    serverId: string | null;
    playerId: string | null;
    cardPoolId: string | null;
    cardPoolType: number | null;
    languageCode: string | null;
    recordId: string | null;
} | null => {
    try {
        const urlObj = new URL(fullUrl.replace('#/record', '')); // 处理掉 hash 部分方便解析
        const params = urlObj.searchParams;

        const gachaType = params.get('gacha_type');

        return {
            serverId: params.get('svr_id'),
            playerId: params.get('player_id'),
            cardPoolId: params.get('resources_id'), // 注意这里：url里叫resources_id，API里叫cardPoolId
            cardPoolType: gachaType ? parseInt(gachaType, 10) : null,
            languageCode: params.get('lang'),
            recordId: params.get('record_id')
        };
    } catch (e) {
        console.error('URL 解析失败', e);
        return null;
    }
};
```
